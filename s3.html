Python & AWS S3
Best Practices

A beginner-friendly guide to building a clean, testable S3 file manager in Python. You'll learn to use an Enum-based bucket type system (SAFE vs TEMP), write a reusable class for uploads and downloads, and test everything with pytest and moto.

Python 3.12+ boto3 ~1.42 pytest 9.x moto 5.x

2 · The BucketType Enum

Using an Enum keeps bucket categories explicit and prevents typos. Each enum member holds a prefix that gets prepended to every bucket name, making it trivial to find and clean up all TEMP buckets later.

src/bucket_type.py
"""Enum that defines the two bucket categories."""

from enum import Enum


class BucketType(Enum):
    """
    SAFE  – long-lived storage, never auto-deleted.
    TEMP  – short-lived storage, cleaned up after use.

    Each member's value is the prefix added to bucket names
    so you can easily list / filter buckets by type.
    """

    SAFE = "safe-"
    TEMP = "temp-"

    def make_bucket_name(self, name: str) -> str:
        """Build the full bucket name: prefix + user-supplied name."""
        return f"{self.value}{name}"

💡 Why an Enum? — Enums are safer than plain strings. Your IDE will autocomplete BucketType.TEMP, and a typo like BucketType.TMP will raise an AttributeError immediately instead of silently creating the wrong bucket.

3 · The S3BucketManager Class

This class wraps all S3 operations behind a clean interface. It accepts a BucketType and a name, then handles upload, download, listing, and cleanup.

src/s3_manager.py
"""Reusable S3 bucket manager with SAFE / TEMP support."""

from __future__ import annotations

import logging
from pathlib import Path

import boto3
from botocore.exceptions import ClientError

from .bucket_type import BucketType

logger = logging.getLogger(__name__)


class S3BucketManager:
    """Manage uploads, downloads, and cleanup for a single S3 bucket.

    Args:
        bucket_type: Whether this is a SAFE or TEMP bucket.
        name:        A short, descriptive name (e.g. "reports", "uploads").
        region:      AWS region for the bucket.  Defaults to us-east-1.
    """

    def __init__(
        self,
        bucket_type: BucketType,
        name: str,
        region: str = "us-east-1",
    ) -> None:
        self.bucket_type = bucket_type
        self.bucket_name = bucket_type.make_bucket_name(name)
        self.region = region
        self._client = boto3.client("s3", region_name=region)

    # ── Bucket lifecycle ──────────────────────────────────

    def create_bucket(self) -> None:
        """Create the S3 bucket if it does not already exist."""
        try:
            if self.region == "us-east-1":
                self._client.create_bucket(Bucket=self.bucket_name)
            else:
                self._client.create_bucket(
                    Bucket=self.bucket_name,
                    CreateBucketConfiguration={
                        "LocationConstraint": self.region,
                    },
                )
            logger.info("Created bucket %s", self.bucket_name)
        except ClientError as exc:
            if exc.response["Error"]["Code"] == "BucketAlreadyOwnedByYou":
                logger.info("Bucket %s already exists", self.bucket_name)
            else:
                raise

    # ── Upload ────────────────────────────────────────────

    def upload_file(
        self, local_path: str | Path, s3_key: str | None = None,
    ) -> str:
        """Upload a local file to the bucket.

        Args:
            local_path: Path to the file on disk.
            s3_key:     Object key in S3.  Defaults to the filename.

        Returns:
            The S3 key that was written.
        """
        local_path = Path(local_path)
        if not local_path.is_file():
            raise FileNotFoundError(f"No such file: {local_path}")

        key = s3_key or local_path.name
        self._client.upload_file(
            Filename=str(local_path),
            Bucket=self.bucket_name,
            Key=key,
        )
        logger.info("Uploaded %s → s3://%s/%s", local_path, self.bucket_name, key)
        return key

    def upload_bytes(
        self, data: bytes, s3_key: str,
    ) -> str:
        """Upload raw bytes directly to S3."""
        self._client.put_object(
            Bucket=self.bucket_name, Key=s3_key, Body=data,
        )
        logger.info("Uploaded %d bytes → s3://%s/%s", len(data), self.bucket_name, s3_key)
        return s3_key

    # ── Download ──────────────────────────────────────────

    def download_file(
        self, s3_key: str, local_path: str | Path,
    ) -> Path:
        """Download an S3 object to a local file.

        Returns:
            The Path of the downloaded file.
        """
        local_path = Path(local_path)
        local_path.parent.mkdir(parents=True, exist_ok=True)
        self._client.download_file(
            Bucket=self.bucket_name, Key=s3_key, Filename=str(local_path),
        )
        logger.info("Downloaded s3://%s/%s → %s", self.bucket_name, s3_key, local_path)
        return local_path

    def download_bytes(self, s3_key: str) -> bytes:
        """Download an S3 object as raw bytes."""
        response = self._client.get_object(
            Bucket=self.bucket_name, Key=s3_key,
        )
        return response["Body"].read()

    # ── Listing ───────────────────────────────────────────

    def list_objects(self, prefix: str = "") -> list[str]:
        """Return a list of S3 keys in the bucket."""
        paginator = self._client.get_paginator("list_objects_v2")
        keys: list[str] = []
        for page in paginator.paginate(Bucket=self.bucket_name, Prefix=prefix):
            for obj in page.get("Contents", []):
                keys.append(obj["Key"])
        return keys

    # ── Cleanup (TEMP buckets only) ───────────────────────

    def cleanup(self) -> int:
        """Delete ALL objects in a TEMP bucket.

        Raises:
            ValueError: If called on a SAFE bucket.

        Returns:
            The number of objects deleted.
        """
        if self.bucket_type is not BucketType.TEMP:
            raise ValueError(
                f"cleanup() is only allowed on TEMP buckets, "
                f"got {self.bucket_type.name}"
            )

        keys = self.list_objects()
        if not keys:
            logger.info("Bucket %s is already empty", self.bucket_name)
            return 0

        # S3 delete_objects accepts up to 1 000 keys at a time
        deleted = 0
        for i in range(0, len(keys), 1000):
            batch = keys[i : i + 1000]
            self._client.delete_objects(
                Bucket=self.bucket_name,
                Delete={"Objects": [{"Key": k} for k in batch]},
            )
            deleted += len(batch)

        logger.info("Cleaned up %d objects from %s", deleted, self.bucket_name)
        return deleted

4 · Upload Examples

Here are standalone functions that show how to use the manager in real application code.

examples/upload_examples.py
"""Upload helpers that use S3BucketManager."""

from src.bucket_type import BucketType
from src.s3_manager import S3BucketManager


def upload_report_to_safe_bucket(filepath: str) -> str:
    """Upload a report to the long-lived SAFE bucket."""
    manager = S3BucketManager(BucketType.SAFE, "reports")
    manager.create_bucket()
    return manager.upload_file(filepath)


def upload_csv_to_temp_bucket(filepath: str) -> str:
    """Upload a CSV to a TEMP bucket for processing."""
    manager = S3BucketManager(BucketType.TEMP, "csv-imports")
    manager.create_bucket()
    return manager.upload_file(filepath, s3_key="incoming/data.csv")


def upload_raw_json(payload: bytes) -> str:
    """Upload raw JSON bytes to a TEMP bucket."""
    manager = S3BucketManager(BucketType.TEMP, "api-payloads")
    manager.create_bucket()
    return manager.upload_bytes(payload, s3_key="payload.json")


# ── Quick demo ────────────────────────────────────────
if __name__ == "__main__":
    key = upload_report_to_safe_bucket("quarterly_report.pdf")
    print(f"Uploaded to SAFE bucket with key: {key}")

5 · Download Examples

examples/download_examples.py
"""Download helpers that use S3BucketManager."""

from pathlib import Path
from src.bucket_type import BucketType
from src.s3_manager import S3BucketManager


def download_report(s3_key: str, dest: str = "./downloads") -> Path:
    """Download a report from the SAFE bucket to a local folder."""
    manager = S3BucketManager(BucketType.SAFE, "reports")
    local_path = Path(dest) / s3_key
    return manager.download_file(s3_key, local_path)


def download_temp_csv_as_bytes(s3_key: str) -> bytes:
    """Download a CSV from the TEMP bucket as raw bytes."""
    manager = S3BucketManager(BucketType.TEMP, "csv-imports")
    return manager.download_bytes(s3_key)


def list_and_download_all(bucket_type: BucketType, name: str, dest: str) -> list[Path]:
    """List every object in a bucket, then download them all."""
    manager = S3BucketManager(bucket_type, name)
    paths: list[Path] = []
    for key in manager.list_objects():
        paths.append(manager.download_file(key, Path(dest) / key))
    return paths

6 · TEMP Bucket Cleanup

The cleanup() method deletes every object in a TEMP bucket. It refuses to run on SAFE buckets — a guardrail built right into the class.

examples/cleanup_example.py
"""Example: process a file then clean up the TEMP bucket."""

from src.bucket_type import BucketType
from src.s3_manager import S3BucketManager


def process_and_cleanup(filepath: str) -> None:
    """Upload → process → clean up.  A typical TEMP bucket lifecycle."""
    manager = S3BucketManager(BucketType.TEMP, "processing")
    manager.create_bucket()

    # 1. Upload
    key = manager.upload_file(filepath)

    # 2. Do your processing…
    data = manager.download_bytes(key)
    result = data.upper()  # placeholder for real work
    print(f"Processed {len(result)} bytes")

    # 3. Clean up all temp files
    deleted = manager.cleanup()
    print(f"Removed {deleted} temp objects")


# This will RAISE a ValueError — SAFE buckets can't be cleaned:
# safe_mgr = S3BucketManager(BucketType.SAFE, "reports")
# safe_mgr.cleanup()  → ValueError!

⚠️ Safety guard — Calling cleanup() on a BucketType.SAFE bucket will always raise a ValueError. This prevents accidental data loss.

7 · Pytest Basics

If you're new to pytest, here's the essentials. Pytest discovers any file named test_*.py and runs every function whose name starts with test_.

Writing your first test

tests/test_demo.py
# A plain function — no classes or boilerplate needed

def test_addition():
    assert 1 + 1 == 2


def test_string_upper():
    assert "hello".upper() == "HELLO"

Running tests

terminal
# Run all tests
pytest

# Verbose output (shows each test name)
pytest -v

# Run only tests in one file
pytest tests/test_s3_manager.py

# Run a single test by name
pytest -k "test_upload"

# Stop on first failure
pytest -x

Fixtures — reusable test setup

Fixtures let you share setup code between tests. Define them in conftest.py so every test file can use them automatically.

tests/conftest.py
import pytest


@pytest.fixture
def sample_text():
    """Provide some sample text for tests."""
    return "Hello, S3!"


@pytest.fixture
def tmp_file(tmp_path):
    """Create a temporary file with some content."""
    file = tmp_path / "test_upload.txt"
    file.write_text("some file content")
    return file

Testing for expected errors

test example
import pytest


def test_divide_by_zero():
    """Verify that dividing by zero raises the right error."""
    with pytest.raises(ZeroDivisionError):
        _ = 1 / 0

💡 Pytest tip — Use pytest.raises() to assert that code raises a specific exception. This is how we'll verify that cleanup() rejects SAFE buckets.

8 · Testing S3 Code with Moto

Moto intercepts all boto3 calls and runs them against an in-memory fake AWS. No real AWS account needed, no network calls, and tests run in milliseconds.

🔴 Important — Always set up mocks before creating boto3 clients. The @mock_aws decorator or fixture must be active before any boto3.client() call happens.

Shared fixtures with moto

tests/conftest.py
"""Shared fixtures for all S3 tests."""

import os
import pytest
from moto import mock_aws


@pytest.fixture(autouse=True)
def aws_credentials():
    """Set fake AWS credentials so boto3 never touches real AWS."""
    os.environ["AWS_ACCESS_KEY_ID"] = "testing"
    os.environ["AWS_SECRET_ACCESS_KEY"] = "testing"
    os.environ["AWS_SECURITY_TOKEN"] = "testing"
    os.environ["AWS_SESSION_TOKEN"] = "testing"
    os.environ["AWS_DEFAULT_REGION"] = "us-east-1"


@pytest.fixture
def mock_s3():
    """Activate moto's mock for all AWS services."""
    with mock_aws():
        yield


@pytest.fixture
def tmp_file(tmp_path):
    """Create a small temp file for upload tests."""
    f = tmp_path / "hello.txt"
    f.write_text("hello world")
    return f

Full test suite

tests/test_s3_manager.py
"""Tests for S3BucketManager — runs against moto, no real AWS needed."""

import pytest

from src.bucket_type import BucketType
from src.s3_manager import S3BucketManager


# ── BucketType enum tests ─────────────────────────────

class TestBucketType:
    """Verify the BucketType enum behaviour."""

    def test_safe_prefix(self):
        assert BucketType.SAFE.value == "safe-"

    def test_temp_prefix(self):
        assert BucketType.TEMP.value == "temp-"

    def test_make_bucket_name(self):
        assert BucketType.TEMP.make_bucket_name("uploads") == "temp-uploads"
        assert BucketType.SAFE.make_bucket_name("reports") == "safe-reports"


# ── Upload tests ──────────────────────────────────────

class TestUpload:
    """Test file and byte uploads to both bucket types."""

    def test_upload_file(self, mock_s3, tmp_file):
        manager = S3BucketManager(BucketType.TEMP, "uploads")
        manager.create_bucket()

        key = manager.upload_file(tmp_file)

        assert key == "hello.txt"
        assert "hello.txt" in manager.list_objects()

    def test_upload_file_custom_key(self, mock_s3, tmp_file):
        manager = S3BucketManager(BucketType.SAFE, "docs")
        manager.create_bucket()

        key = manager.upload_file(tmp_file, s3_key="custom/path.txt")

        assert key == "custom/path.txt"

    def test_upload_bytes(self, mock_s3):
        manager = S3BucketManager(BucketType.TEMP, "raw")
        manager.create_bucket()

        key = manager.upload_bytes(b"raw data", "data.bin")

        assert key == "data.bin"
        assert manager.download_bytes("data.bin") == b"raw data"

    def test_upload_missing_file_raises(self, mock_s3):
        manager = S3BucketManager(BucketType.TEMP, "uploads")
        manager.create_bucket()

        with pytest.raises(FileNotFoundError):
            manager.upload_file("/does/not/exist.txt")


# ── Download tests ────────────────────────────────────

class TestDownload:
    """Test downloads to files and bytes."""

    def test_download_file(self, mock_s3, tmp_file, tmp_path):
        manager = S3BucketManager(BucketType.SAFE, "docs")
        manager.create_bucket()
        manager.upload_file(tmp_file)

        dest = tmp_path / "downloaded.txt"
        result = manager.download_file("hello.txt", dest)

        assert result.read_text() == "hello world"

    def test_download_bytes(self, mock_s3):
        manager = S3BucketManager(BucketType.TEMP, "raw")
        manager.create_bucket()
        manager.upload_bytes(b"binary content", "file.bin")

        data = manager.download_bytes("file.bin")

        assert data == b"binary content"


# ── Cleanup tests ─────────────────────────────────────

class TestCleanup:
    """Test the TEMP-only cleanup behaviour."""

    def test_cleanup_deletes_all_objects(self, mock_s3):
        manager = S3BucketManager(BucketType.TEMP, "scratch")
        manager.create_bucket()
        manager.upload_bytes(b"a", "file1.txt")
        manager.upload_bytes(b"b", "file2.txt")
        manager.upload_bytes(b"c", "file3.txt")

        deleted = manager.cleanup()

        assert deleted == 3
        assert manager.list_objects() == []

    def test_cleanup_on_empty_bucket(self, mock_s3):
        manager = S3BucketManager(BucketType.TEMP, "empty")
        manager.create_bucket()

        assert manager.cleanup() == 0

    def test_cleanup_raises_on_safe_bucket(self, mock_s3):
        manager = S3BucketManager(BucketType.SAFE, "important")
        manager.create_bucket()

        with pytest.raises(ValueError, match="only allowed on TEMP"):
            manager.cleanup()


# ── Listing tests ─────────────────────────────────────

class TestListObjects:
    """Test object listing and prefix filtering."""

    def test_list_objects_with_prefix(self, mock_s3):
        manager = S3BucketManager(BucketType.TEMP, "data")
        manager.create_bucket()
        manager.upload_bytes(b"1", "reports/jan.csv")
        manager.upload_bytes(b"2", "reports/feb.csv")
        manager.upload_bytes(b"3", "logs/app.log")

        reports = manager.list_objects(prefix="reports/")

        assert len(reports) == 2
        assert "logs/app.log" not in reports

9 · Best Practices Checklist

🔐 Never hardcode AWS credentials

Use IAM roles, environment variables, or the ~/.aws/credentials file. Never commit secrets to git.

📛 Use Enums for bucket categories

Enums make your code self-documenting. BucketType.TEMP is clearer and safer than the string "temp".

🧪 Test with moto, not real AWS

Moto runs in-memory and needs no AWS account. Tests are fast, free, and repeatable. Always set fake credentials in your fixtures.

🗂 Put fixtures in conftest.py

Shared setup (like mock_s3) goes in conftest.py. Pytest auto-discovers it — no imports needed in test files.

🛡 Guard destructive operations

The cleanup() method checks bucket_type and raises if it's not TEMP. Always add guardrails around delete operations.

📝 Use type hints everywhere

Type hints like str | Path and -> list[str] make your code readable and let tools catch bugs before runtime.

📊 Use logging, not print()

The logging module lets you control verbosity with levels (DEBUG, INFO, WARNING). Print statements get messy fast.

🔄 Use paginators for listing

S3 returns at most 1,000 objects per request. Always use get_paginator("list_objects_v2") to handle buckets with many files.
